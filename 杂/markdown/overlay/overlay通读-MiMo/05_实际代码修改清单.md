### 修改2: 重写 draw_manager_text.cc 中的 DRW_text_cache_draw()

**文件**: `source/blender/draw/intern/draw_manager_text.cc`

**替换** `void DRW_text_cache_draw(const DRWTextStore *dt, ...)` 函数为：

```cpp
void DRW_text_cache_draw(const DRWTextStore *dt,
                         const ARegion *region,
                         const View3D *v3d,
                         blender::gpu::Texture *depth_tx,
                         const uchar alpha)
{
  if (v3d == nullptr) {
    return;
  }

  RegionView3D *rv3d = (RegionView3D *)region->regiondata;
  int tot = 0;

  /* 保存当前帧缓冲状态 */
  blender::gpu::FrameBuffer *original_fb = GPU_framebuffer_active_get();

  /* 创建深度采样帧缓冲 */
  blender::gpu::FrameBuffer *depth_fb = nullptr;
  if (depth_tx) {
    depth_fb = GPU_framebuffer_create("text_depth_read");
    GPU_framebuffer_texture_attach(depth_fb, depth_tx, 0, 0);
    GPU_framebuffer_bind(depth_fb);
  }

  /* 遍历所有文本 */
  BLI_memiter_handle it;
  BLI_memiter_iter_init(dt->cache_strings, &it);
  int text_index = 0;

  while (ViewCachedString *vos = (ViewCachedString *)BLI_memiter_iter_step(&it)) {
    text_index++;

    /* 2D投影测试 */
    float screen_pos[3];
    eV3DProjStatus proj_status = ED_view3d_project_float_ex(
        region, vos->vec, screen_pos, rv3d->perspmat, V3D_PROJ_TEST_CLIP_DEFAULT);

    if (proj_status == V3D_PROJ_RET_OK) {
      bool is_visible = true;

      /* ⭐ 深度遮挡测试 */
      if (depth_fb) {
        int x = int(screen_pos[0]);
        int y = int(screen_pos[1]);

        /* 边界检查 */
        if (x >= 0 && x < region->winx && y >= 0 && y < region->winy) {
          float scene_depth;
          GPU_framebuffer_read_depth(depth_fb, x, y, 1, 1, GPU_DATA_FLOAT, &scene_depth);

          /* NDC深度转换 [-1,1] -> [0,1] */
          float text_depth = screen_pos[2] * 0.5f + 0.5f;

          /* 深度比较: 文本在场景前方才可见 */
          is_visible = (text_depth < scene_depth + 0.00001f);
        }
      }

      if (is_visible) {
        vos->sco[0] = short(screen_pos[0]);
        vos->sco[1] = short(screen_pos[1]);
        tot++;
      }
      else {
        vos->sco[0] = IS_CLIPPED;  /* 标记为被遮挡 */
      }
    }
    else {
      vos->sco[0] = IS_CLIPPED;
    }
  }

  /* 恢复状态 */
  if (depth_fb) {
    GPU_framebuffer_free(depth_fb);
  }
  if (original_fb) {
    GPU_framebuffer_bind(original_fb);
  }

  /* 只绘制可见文本 */
  if (tot > 0) {
    drw_text_cache_draw_ex(dt, region, alpha);
  }
}
```

**说明**:
- 如果函数已存在但逻辑不全，用此覆盖
- 如果完全不存在，添加到文件末尾
- 需要包含头文件: `#include "GPU_framebuffer.hh"`

```cpp
void Instance::draw_text(Framebuffer &framebuffer)
{
  if (state.show_text == false) {
    return;
  }

  GPU_framebuffer_bind(framebuffer);

  /* ⭐ 决定是否启用深度测试 */
  bool use_depth_test = true;

  /* X-ray模式特殊处理 */
  if (state.v3d) {
    const short xray_mask = V3D_SHADING_XRAY | V3D_SHADING_XRAY_WIREFRAME;
    if (state.v3d->shading.flag & xray_mask) {
      float alpha = 1.0f;
#if BLENDER_VERSION >= 400
      alpha = state.v3d->shading.xray_alpha;
#else
      alpha = state.v3d->shading.xray_alpha_z;
#endif
      /* X-ray alpha > 0.9 时才进行深度测试 */
      if (alpha < 0.9f) {
        use_depth_test = false;
      }
    }
  }

  /* ⭐ 传递深度纹理 */
  blender::gpu::Texture *depth_tx = nullptr;
  if (use_depth_test && resources.depth_tx.is_valid()) {
    depth_tx = resources.depth_tx;
  }

  /* 调用文本绘制 */
  DRW_text_cache_draw(state.dt, state.region, state.v3d, depth_tx, 255);
}
```

```cpp
gpu::Batch *DRW_cache_mesh_points_viewer_attribute_get(Object *ob)
{
  Mesh *mesh = DRW_object_get_data_for_drawing<Mesh>(*ob);

  /* 检查属性存在 */
  if (!mesh->attributes().contains(".viewer")) {
    return nullptr;
  }

  /* 获取点位置和属性 */
  const PositionAttribute positions = mesh->attributes().get_position();
  const VArray attribute = *mesh->attributes().lookup<ColorGeometry4f>(".viewer");

  /* 创建/获取VBO */
  gpu::VertBuf *vbo = mesh->runtime->evaluated_cache.vert_buf;
  if (vbo == nullptr) {
    vbo = GPU_vertbuf_create_with_format(format_pos_col);
    GPU_vertbuf_data_alloc(*vbo, mesh->totvert);

    for (int i = 0; i < mesh->totvert; i++) {
      vbo->data[i].pos = positions[i];
      vbo->data[i].col = attribute[i];
    }
  }

  /* 创建batch */
  gpu::Batch *batch = GPU_batch_create(GPU_PRIM_POINTS, vbo, nullptr);
  return batch;
}
```

```glsl
bool test_occlusion(float4 clip_pos)
{
  /* 简单深度测试，可选 */
  float3 ndc = clip_pos.xyz / clip_pos.w;
  ndc = ndc * 0.5f + 0.5f;
  float scene_depth = texture(depth_tx, ndc.xy).r;

void main()
{
  /* 避免未定义行为 */

  /* 每条线2个顶点 */
  int vertex_id = gl_VertexID;
  int vector_id = vertex_id / 2;
  bool is_start = (vertex_id & 1) == 0;

  /* 读取原始数据 */
```

### 修改5: 实现批次生成函数

```cpp
/* 矢量批次生成 */
gpu::Batch *DRW_cache_mesh_vector_attributes_get(Object *ob, const char *attr_name)
{
  /* 策略:
     1. 读取位置 + 矢量属性
     2. 创建LINES原语的顶点缓冲
       - 顶点0 = start_z
       - 顶点1 = start + vector
     3. 每2个顶点组成一条线
  */

  Mesh *mesh = DRW_object_get_data_for_drawing<Mesh>(*ob);
  auto positions = mesh->attributes().get_position();
  auto vectors = mesh->attributes().lookup<float3>(attr_name);

  if (!vectors) {
    return nullptr;  /* 没有矢量数据 */
  }

  /* 创建格式: position + color (矢量可以用颜色编码) */
  GPUVertFormat format = {0};
  GPU_vertformat_attr_add(&format, "pos", GPU_COMP_F32, 3, GPU_FETCH_FLOAT);
  GPU_vertformat_attr_add(&format, "color", GPU_COMP_F32, 4, GPU_FETCH_FLOAT);

  gpu::VertBuf *vbo = GPU_vertbuf_create_with_format(&format);

  /* 每个矢量需要2个顶点 */
  int vertex_count = vectors->size() * 2;
  GPU_vertbuf_data_alloc(*vbo, vertex_count);

  for (int i = 0; i < vectors->size(); i++) {
    int base = i * 2;

    /* 起点 */
    vbo->data[base].pos = positions[i];
    vbo->data[base].color = float4(0.2, 0.8, 0.2, 1.0);  /* 绿色 */

    /* 终点 */
    vbo->data[base + 1].pos = positions[i] + (*vectors)[i] * 0.1f;  /* 缩放 */
    vbo->data[base + 1].color = float4(1.0, 0.0, 0.0, 1.0);  /* 红色 */
  }

  gpu::Batch *batch = GPU_batch_create(GPU_PRIM_LINES, vbo, nullptr);
  return batch;
}
```
