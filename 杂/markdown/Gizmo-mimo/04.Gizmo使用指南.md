## 概述

本章介绍如何在 Blender 中实际使用 gizmo 系统，包括创建、配置、绑定和生命周期管理。

<span style="background-color: #2E7D32; color: white; padding: 2px 6px; border-radius: 3px;">★ Insight</span>
所有 gizmo 操作都通过 `source/blender/windowmanager/gizmo/WM_gizmo_api.hh` 中的 API 进行。

## 2. 基本使用流程

### 2.1. 创建 Gizmo

**C API**:
```cpp
// 方法 1: 使用字符串 ID
wmGizmo *gz = WM_gizmo_new("GIZMO_GT_arrow_3d", gzgroup, nullptr);

// 方法 2: 使用类型指针（更高效）
const wmGizmoType *gzt = WM_gizmotype_find("GIZMO_GT_arrow_3d", false);
wmGizmo *gz = WM_gizmo_new_ptr(gzt, gzgroup, nullptr);
```

**源码位置**: `source/blender/editors/include/ED_gizmo_library.hh:51-57`

**参数说明**:
- `"GIZMO_GT_arrow_3d"`: gizmo 类型名称
- `gzgroup`: 父 gizmo 组
- `nullptr`: 可选的初始属性

---

### 2.2. 配置属性

**设置样式**:
```cpp
// 箭头样式
RNA_enum_set(gz->ptr, "draw_style", ED_GIZMO_ARROW_STYLE_BOX);

// 绘制选项
RNA_enum_set(gz->ptr, "draw_options",
  ED_GIZMO_ARROW_DRAW_FLAG_STEM |
  ED_GIZMO_ARROW_DRAW_FLAG_ORIGIN);
```

**设置尺寸**:
```cpp
// 边界框尺寸
float dims[2] = {100.0f, 100.0f};
RNA_float_set_array(gz->ptr, "dimensions", dims);

// 基础缩放
gz->scale_basis = 1.0f;

// 线宽
gz->line_width = 2.0f;
```

**设置颜色**:
```cpp
float color[4] = {1.0f, 0.5f, 0.0f, 0.8f};  // RGBA
WM_gizmo_set_color(gz, color);

float color_hi[4] = {1.0f, 1.0f, 0.0f, 1.0f};
WM_gizmo_set_color_highlight(gz, color_hi);
```

static void gizmo_node_backdrop_prop_matrix_get(const wmGizmo *gz,
                                                wmGizmoProperty *gz_prop,

{
  float (*matrix)[4] = (float (*)[4])value_p;

## 3. Gizmo Group 生命周期

### 3.1. Poll - 条件检查

**作用**: 决定是否显示 gizmo 组

```cpp
static bool WIDGETGROUP_my_poll(const bContext *C, wmGizmoGroupType * /*gzgt*/)
{
  SpaceNode *snode = CTX_wm_space_node(C);
  if (snode == nullptr) {
    return false;
  }

  // 检查是否在 compositor
  if (!snode->edittree || snode->edittree->type != NTREE_COMPOSIT) {
    return false;
  }

  // 检查是否有活动节点
  bNode *node = bke::node_get_active(*snode->edittree);
  if (!node) {
    return false;
  }

  // 检查节点类型
  return node->is_type("CompositorNodeMyType");
}
```

**源码位置**: `source/blender/editors/space_node/node_gizmo.cc:134-148`

---

### 3.2. Setup - 初始化

**作用**: 创建 gizmo，设置永久数据

```cpp
static void WIDGETGROUP_my_setup(const bContext * /*C*/, wmGizmoGroup *gzgroup)
{
  // 分配自定义数据
  MyGizmoData *data = MEM_new<MyGizmoData>(__func__);

  // 创建 gizmo
  data->gizmo = WM_gizmo_new("GIZMO_GT_cage_2d", gzgroup, nullptr);

  // 配置
  RNA_enum_set(data->gizmo->ptr, "transform",
               ED_GIZMO_CAGE_XFORM_FLAG_TRANSLATE |
               ED_GIZMO_CAGE_XFORM_FLAG_SCALE);

  // 设置自定义数据
  gzgroup->customdata = data;

  // 设置释放函数
  gzgroup->customdata_free = [](void *customdata) {
    MEM_delete(static_cast<MyGizmoData *>(customdata));
  };
}
```

**源码位置**: `source/blender/editors/space_node/node_gizmo.cc:150-161`

### 3.3. Refresh - 数据更新

**作用**: 根据当前数据更新 gizmo

```cpp
static void WIDGETGROUP_my_refresh(const bContext *C, wmGizmoGroup *gzgroup)
{
  MyGizmoData *data = (MyGizmoData *)gzgroup->customdata;
  wmGizmo *gz = data->gizmo;

  // 获取上下文数据
  SpaceNode *snode = CTX_wm_space_node(C);
  bNode *node = bke::node_get_active(*snode->edittree);

  // 检查是否需要隐藏
  if (!node || !node->is_type("CompositorNodeMyType")) {
    WM_gizmo_set_flag(gz, WM_GIZMO_HIDDEN, true);
    return;
  }

  // 更新尺寸
  float dims[2] = {node->width, node->height};
  RNA_float_set_array(gz->ptr, "dimensions", dims);

  // 更新位置
  float origin[3] = {node->locx, node->locy, 0.0f};
  WM_gizmo_set_matrix_location(gz, origin);

  // 显示
  WM_gizmo_set_flag(gz, WM_GIZMO_HIDDEN, false);

  // 绑定属性
  PointerRNA nodeptr = RNA_pointer_create_discrete(
      reinterpret_cast<ID *>(snode->edittree),
      &RNA_NodeSocket,
      node->inputs.first);
  WM_gizmo_target_property_def_rna(gz, "offset", &nodeptr, "default_value", -1);
}
```

**源码位置**: `source/blender/editors/space_node/node_gizmo.cc:163-203`

### 3.4. Draw Prepare - 绘制准备

**作用**: 每次绘制前的最终调整

```cpp
static void WIDGETGROUP_my_draw_prepare(const bContext *C, wmGizmoGroup *gzgroup)
{
  ARegion *region = CTX_wm_region(C);
  wmGizmo *gz = (wmGizmo *)gzgroup->gizmos.first;
  SpaceNode *snode = CTX_wm_space_node(C);

  // 计算空间矩阵
  float matrix_space[4][4];
  unit_m4(matrix_space);
  mul_v3_fl(matrix_space[0], snode->zoom);
  mul_v3_fl(matrix_space[1], snode->zoom);
  matrix_space[3][0] = (region->winx / 2) + snode->xof;
  matrix_space[3][1] = (region->winy / 2) + snode->yof;

  copy_m4_m4(gz->matrix_space, matrix_space);
}
```

### 4.2. 自定义函数绑定

**适用场景**:
- 属性不在 RNA 中
- 需要复杂的转换逻辑
- 需要访问多个属性

**示例 - 矩阵绑定**:
```cpp
// Getter: 从数据结构读取到 gizmo
static void my_gizmo_get(const wmGizmo *gz,
                         wmGizmoProperty *gz_prop,
                         void *value_p)
{
  float (*matrix)[4] = (float (*)[4])value_p;
  MyData *data = (MyData *)gz_prop->custom_func.user_data;

  // 从 data 填充 matrix
  matrix[0][0] = data->scale_x;
  matrix[1][1] = data->scale_y;
  matrix[3][0] = data->pos_x;
  matrix[3][1] = data->pos_y;
}

// Setter: 从 gizmo 写入到数据结构
static void my_gizmo_set(const wmGizmo *gz,
                         wmGizmoProperty *gz_prop,
                         const void *value_p)
{
  const float (*matrix)[4] = (const float (*)[4])value_p;
  MyData *data = (MyData *)gz_prop->custom_func.user_data;

  // 从 matrix 写入 data
  data->scale_x = matrix[0][0];
  data->scale_y = matrix[1][1];
  data->pos_x = matrix[3][0];
  data->pos_y = matrix[3][1];

  // 触发更新
  my_data_update(data);
}

// 绑定
wmGizmoPropertyFnParams params{};
params.value_get_fn = my_gizmo_get;
params.value_set_fn = my_gizmo_set;
params.user_data = my_data;
WM_gizmo_target_property_def_func(gz, "matrix", ¶ms);
```

**源码位置**: `source/blender/editors/include/ED_gizmo_library.hh:245-247`

## 5. 实际使用示例

### 5.1. 简单箭头 gizmo

```cpp
// 1. 创建组类型
void MY_GT_arrow_group(wmGizmoGroupType *gzgt)
{
  gzgt->name = "My Arrow Gizmo";
  gzgt->idname = "MY_GT_arrow_group";
  gzgt->flag = WM_GIZMOGROUPTYPE_PERSISTENT;
  gzgt->poll = my_arrow_poll;
  gzgt->setup = my_arrow_setup;
  gzgt->refresh = my_arrow_refresh;
  gzgt->setup_keymap = WM_gizmogroup_setup_keymap_generic_maybe_drag;
}

// 2. Poll
static bool my_arrow_poll(const bContext *C, wmGizmoGroupType * /*gzgt*/)
{
  Object *ob = CTX_data_active_object(C);
  return (ob && ob->type == OB_LIGHT);
}

// 3. Setup
static void my_arrow_setup(const bContext * /*C*/, wmGizmoGroup *gzgroup)
{
  MyArrowData *data = MEM_new<MyArrowData>(__func__);
  data->gz = WM_gizmo_new("GIZMO_GT_arrow_3d", gzgroup, nullptr);

  // 配置
  RNA_enum_set(data->gz->ptr, "draw_style", ED_GIZMO_ARROW_STYLE_BOX);
  data->gz->scale_basis = 0.5f;

  gzgroup->customdata = data;
  gzgroup->customdata_free = [](void *d) {
    MEM_delete(static_cast<MyArrowData *>(d));
  };
}

// 4. Refresh
static void my_arrow_refresh(const bContext *C, wmGizmoGroup *gzgroup)
{
  MyArrowData *data = (MyArrowData *)gzgroup->customdata;
  Object *ob = CTX_data_active_object(C);

  if (!ob || ob->type != OB_LIGHT) {
    WM_gizmo_set_flag(data->gz, WM_GIZMO_HIDDEN, true);
    return;
  }

  // 设置位置
  float pos[3] = {ob->loc[0], ob->loc[1] + 2.0f, ob->loc[2]};
  WM_gizmo_set_matrix_location(data->gz, pos);

  // 绑定到能量
  PointerRNA ptr = RNA_pointer_create_discrete(&ob->id, &RNA_Light, ob->data);
  WM_gizmo_target_property_def_rna(data->gz, "offset", &ptr, "energy", -1);

  WM_gizmo_set_flag(data->gz, WM_GIZMO_HIDDEN, false);
}
```

// 角钉 gizmo - 4 个移动点
static void corner_pin_setup(const bContext * /*C*/, wmGizmoGroup *gzgroup)
{
  CornerPinData *data = MEM_new<CornerPinData>(__func__);

{
  CornerPinData *data = (CornerPinData *)gzgroup->customdata;

  // 计算空间矩阵

}

## 6. Python API

Blender 也支持通过 Python 创建 gizmo：

```python
import bpy

class MyGizmoGroup(bpy.types.GizmoGroup):
    bl_idname = "MY_GT_custom_group"
    bl_label = "Custom Gizmos"
    bl_options = {'3D', 'PERSISTENT'}

    @classmethod
    def poll(cls, context):
        return context.object is not None

    def setup(self, context):
        # 创建 gizmo
        gz = self.gizmos.new("GIZMO_GT_arrow_3d")

        # 配置
        gz.draw_style = 'BOX'
        gz.color = (1.0, 0.5, 0.0)
        gz.alpha = 0.5

        # 绑定属性
        gz.target_set_prop("offset", context.object.data, "energy")

        # 设置变换
        gz.matrix_basis[0][0] = 2.0  # X 缩放
        gz.matrix_basis[1][1] = 2.0  # Y 缩放

    def refresh(self, context):
        # 更新位置
        ob = context.object
        loc = (ob.location.x, ob.location.y + 2.0, ob.location.z)
        self.gizmos[0].matrix_basis.translation = loc

# 注册
bpy.utils.register_class(MyGizmoGroup)
```

### 使用流程速查

```cpp
// 1. 创建组类型
void MY_GT_my_group(wmGizmoGroupType *gzgt) {
  gzgt->idname = "MY_GT_my_group";
  gzgt->flag = WM_GIZMOGROUPTYPE_PERSISTENT;
  gzgt->poll = my_poll;
  gzgt->setup = my_setup;
  gzgt->refresh = my_refresh;
  gzgt->setup_keymap = WM_gizmogroup_setup_keymap_generic_maybe_drag;
}

// 2. Setup - 创建 gizmo
static void my_setup(const bContext *C, wmGizmoGroup *gzgroup) {
  MyData *data = MEM_new<MyData>(__func__);
  data->gz = WM_gizmo_new("GIZMO_GT_cage_2d", gzgroup, nullptr);
  RNA_enum_set(data->gz->ptr, "transform", ED_GIZMO_CAGE_XFORM_FLAG_TRANSLATE);
  gzgroup->customdata = data;
}

// 3. Refresh - 更新数据
static void my_refresh(const bContext *C, wmGizmoGroup *gzgroup) {
  MyData *data = (MyData *)gzgroup->customdata;
  // 更新位置、尺寸
  WM_gizmo_set_matrix_location(data->gz, pos);
  // 绑定属性
  WM_gizmo_target_property_def_rna(data->gz, "offset", &ptr, "prop", -1);
}

// 4. Draw Prepare - 空间变换
static void my_draw_prepare(const bContext *C, wmGizmoGroup *gzgroup) {
  // 计算 matrix_space
  copy_m4_m4(gz->matrix_space, matrix_space);
}
```

---

**关键要点**:
1. ✅ 使用 `WM_gizmo_new()` 创建 gizmo
2. ✅ 在 `setup()` 中配置永久属性
3. ✅ 在 `refresh()` 中更新数据和绑定
4. ✅ 在 `draw_prepare()` 中设置空间矩阵
5. ✅ 使用 `WM_gizmo_target_property_def_*()` 绑定属性
6. ✅ 设置 `setup_keymap` 处理输入
