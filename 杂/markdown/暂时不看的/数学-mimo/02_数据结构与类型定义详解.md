```cpp
struct wmGizmoProperty {
  const wmGizmoPropertyType *type = nullptr;  // <span style="color:#FF9800;">属性类型描述</span>
```

#### <span style="background:#4CAF50; color:white; padding:2px 6px;">字段详解</span>

| 字段 | 类型 | 作用 | 示例值 |
|------|------|------|--------|
| `type` | `const wmGizmoPropertyType*` | 属性类型元数据 | 指向包含array_length=16的类型 |

## <span style="color:#FF6B6B; font-size:24px;">3. 矩阵表示法</span>

### <span style="color:#00E676; font-size:18px;">3.1. 4x4齐次矩阵</span>

#### <span style="background:#2196F3; color:white; padding:2px 6px;">C++数组声明</span>

```cpp
// <span style="color:#FF9800;">方法1: 栈上数组</span>
float matrix[4][4];

// <span style="color:#FF9800;">方法2: 指向数组的指针</span>
float (*matrix)[4];

// <span style="color:#FF9800;">方法3: 一维数组模拟</span>
float matrix[16];
// 访问: matrix[row * 4 + col]
```

#### <span style="background:#4CAF50; color:white; padding:2px 6px;">内存布局</span>

```cpp
// <span style="color:#00E676;">row-major 顺序存储</span>
float mat[4][4] = {
  { m00, m01, m02, m03 },  // <span style="color:#FFD93D;">第0行: 旋转/缩放</span>
  { m10, m11, m12, m13 },  // <span style="color:#FFD93D;">第1行: 旋转/缩放</span>
  { m20, m21, m22, m23 },  // <span style="color:#FFD93D;">第2行: 旋转/缩放</span>
  { m30, m31, m32, m33 }   // <span style="color:#FFD93D;">第3行: 平移+透视</span>
};

// <span style="color:#00B4D8;">齐次坐标变换</span>
// [x', y', z', w'] = [x, y, z, 1] × mat
```

#### <span style="background:#E91E63; color:white; padding:2px 6px;">4x4矩阵的用途</span>

| 区域 | 功能 | 典型值 |
|------|------|--------|
| `mat[0..2][0..2]` | 旋转 + 缩放 | 正交矩阵 |
| `mat[0..2][3]` | 平移 | 位置向量 |
| `mat[3][0..2]` | 透视 (通常为0) | 0, 0, 0 |
| `mat[3][3]` | 齐次坐标 | 1 |

### <span style="color:#00E676; font-size:18px;">3.2. 3x3旋转矩阵</span>

#### <span style="background:#2196F3; color:white; padding:2px 6px;">纯旋转矩阵性质</span>

```cpp
// <span style="color:#00E676;">正交矩阵 (Orthogonal Matrix)</span>
// 1. 转置 = 逆: R^T = R^-1
// 2. 行列式 = 1: det(R) = 1
// 3. 行/列向量是单位向量且互相垂直

float rot[3][3] = {
  { r00, r01, r02 },  // <span style="color:#FFD93D;">X轴基向量</span>
  { r10, r11, r12 },  // <span style="color:#FFD93D;">Y轴基向量</span>
  { r20, r21, r22 }   // <span style="color:#FFD93D;">Z轴基向量</span>
};
```

#### <span style="background:#4CAF50; color:white; padding:2px 6px;">单位矩阵</span>

```cpp
float identity[3][3] = {
  { 1, 0, 0 },
  { 0, 1, 0 },
  { 0, 0, 1 }
};

// <span style="color:#00B4D8;">表示: 无旋转，无缩放</span>
```

## <span style="color:#FF6B6B; font-size:24px;">4. 指针类型详解</span>

### <span style="color:#FFD93D; font-size:18px;">4.1. 基础指针类型</span>

```cpp
// <span style="color:#4CAF50;">单个值</span>
int *p1;              // 指向int的指针
float *p2;            // 指向float的指针

// <span style="color:#4CAF50;">数组</span>
int arr[5];           // 5个int的数组
int *p3 = arr;        // 指向数组首元素

// <span style="color:#4CAF50;">指针的指针</span>
int **p4;             // 指向int指针的指针
```

---

### <span style="color:#FFD93D; font-size:18px;">4.2. 复杂指针类型</span>

#### <span style="background:#2196F3; color:white; padding:2px 6px;">指向数组的指针</span>

```cpp
// <span style="color:#00E676;">语法: 类型 (*指针名)[数组大小]</span>

float (*matrix)[4];   // <span style="color:#4CAF50;">指向包含4个float的数组的指针</span>

// <span style="color:#00B4D8;">使用示例</span>
float mat[4][4] = { /* ... */ };
matrix = mat;         // matrix 指向 mat[0]

// <span style="color:#FF5252;">访问方式</span>
matrix[0][0] = 1.0f;  // mat[0][0]
matrix[1][2] = 2.0f;  // mat[1][2]
matrix[3][3] = 3.0f;  // mat[3][3]
```

#### <span style="background:#9C27B0; color:white; padding:2px 6px;">为什么不是这些？</span>

```cpp
// <span style="color:#FF5252;">错误1: 指向指针的指针</span>
float **wrong1;
// 无法使用 matrix[i][j] 访问

// <span style="color:#FF5252;">错误2: 指针数组</span>
float *wrong2[4];
// 这是4个独立的指针，不是连续的4x4

// <span style="color:#FF5252;">错误3: 一维指针</span>
float *wrong3;
// 无法直观地进行二维访问
```

```cpp
#  define BLI_assert(expr) \
```

```cpp
if (gz_prop->type->array_length != 16) {
```

- **发布版本**: 宏被替换为空，无性能开销
- **调试**: 快速发现逻辑错误

### <span style="color:#00E676; font-size:18px;">6.2. BLI_ASSERT_UNIT_M3</span>

**定义位置**: `source/blender/blenlib/BLI_math_matrix.h`

```cpp
// <span style="color:#00E676;">检查矩阵是否为单位矩阵（正交且行列式为1）</span>
#define BLI_ASSERT_UNIT_M3(m) \
  BLI_assert(BLI_matrix_is_orthogonal(m, 3, 0.0001f))
```

#### <span style="background:#2196F3; color:white; padding:2px 6px;">实现细节</span>

```cpp
bool BLI_matrix_is_orthogonal(const float m[3][3],
                              int size,
                              float epsilon)
{
  // <span style="color:#00E676;">检查行向量是否单位长度</span>
  for (int i = 0; i < size; i++) {
    float len = len_v3(m[i]);
    if (fabsf(len - 1.0f) > epsilon) {
      return false;
    }
  }

  // <span style="color:#00E676;">检查行向量是否互相垂直</span>
  for (int i = 0; i < size; i++) {
    for (int j = i + 1; j < size; j++) {
      float dot = dot_v3(m[i], m[j]);
      if (fabsf(dot) > epsilon) {
        return false;
      }
    }
  }

  return true;
}
```

```cpp
mat[0][0] = 1.0f;          // 第0行第0列
mat[i][j] = value;         // 第i行第j列

// <span style="color:#00E676;">作为指针传递</span>
func(mat);                 // 传递给 func(float[4][4])
func((float*)mat);         // 转换为 float*
```
