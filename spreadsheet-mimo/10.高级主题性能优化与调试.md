```cpp
template<typename T>
class ObjectPool {
 private:
  struct Block {
    static const size_t SIZE = 64;
    T objects[SIZE];
    bool used[SIZE];
    Block *next = nullptr;

    Block() {
      memset(used, 0, sizeof(used));
    }
  };

  Block *head_ = nullptr;
  size_t total_allocated_ = 0;
  size_t total_used_ = 0;

 public:
  ObjectPool() = default;
  ~ObjectPool() {
    clear();
  }

  T *alloc()
  {
    // 查找空闲对象
    Block *block = head_;
    while (block) {
      for (size_t i = 0; i < Block::SIZE; i++) {
        if (!block->used[i]) {
          block->used[i] = true;
          total_used_++;
          return &block->objects[i];
        }
      }
      block = block->next;
    }

    // 创建新块
    Block *new_block = new Block();
    new_block->next = head_;
    head_ = new_block;
    total_allocated_ += Block::SIZE * sizeof(T);

    new_block->used[0] = true;
    total_used_++;
    return &new_block->objects[0];
  }

  void free(T *ptr)
  {
    Block *block = head_;
    while (block) {
      // 检查指针是否在当前块中
      if (ptr >= block->objects && ptr < block->objects + Block::SIZE) {
        size_t index = ptr - block->objects;
        if (block->used[index]) {
          block->used[index] = false;
          total_used_--;
        }
        return;
      }
      block = block->next;
    }
  }

  void clear()
  {
    Block *block = head_;
    while (block) {
      Block *next = block->next;
      delete block;
      block = next;
    }
    head_ = nullptr;
    total_allocated_ = 0;
    total_used_ = 0;
  }

  size_t get_total_allocated() const { return total_allocated_; }
  size_t get_total_used() const { return total_used_; }
  size_t get_utilization() const {
    return total_allocated_ > 0 ? (total_used_ * 100) / (total_allocated_ / sizeof(T)) : 0;
  }
};

// 使用示例：过滤器池
class FilterPool {
 private:
  ObjectPool<SpreadsheetRowFilter> pool_;

 public:
  SpreadsheetRowFilter *create()
  {
    SpreadsheetRowFilter *filter = pool_.alloc();
    // 初始化
    memset(filter, 0, sizeof(SpreadsheetRowFilter));
    filter->flag = SPREADSHEET_ROW_FILTER_ENABLED;
    filter->threshold = 0.001f;
    return filter;
  }

  void destroy(SpreadsheetRowFilter *filter)
  {
    // 释放字符串
    if (filter->value_string) {
      MEM_freeN(filter->value_string);
      filter->value_string = nullptr;
    }
    pool_.free(filter);
  }

  void print_stats()
  {
    printf("Filter Pool: %zu allocated, %zu used, %zu%% utilization\n",
           pool_.get_total_allocated() / sizeof(SpreadsheetRowFilter),
           pool_.get_total_used(),
           pool_.get_utilization());
  }
};
```

### 2.2. 对象复用

#### 2.2.1. 对象池模式

```cpp
class ColumnValuesCache {
 private:
  struct CacheEntry {
    std::unique_ptr<ColumnValues> values;
    uint64_t last_used;
    bool in_use;
  };

  std::unordered_map<std::string, CacheEntry> cache_;
  size_t max_size_ = 50;
  uint64_t access_counter_ = 0;

 public:
  // 获取或创建
  ColumnValues *get_or_create(const std::string &column_name,
                              std::function<std::unique_ptr<ColumnValues>()> creator)
  {
    access_counter_++;

    auto it = cache_.find(column_name);
    if (it != cache_.end()) {
      // 更新使用时间
      it->second.last_used = access_counter_;
      it->second.in_use = true;
      return it->second.values.get();
    }

    // 创建新对象
    auto values = creator();

    // 检查容量
    if (cache_.size() >= max_size_) {
      evict_oldest();
    }

    // 插入缓存
    cache_[column_name] = {
      std::move(values),
      access_counter_,
      true
    };

    return cache_[column_name].values.get();
  }

  void release(const std::string &column_name)
  {
    auto it = cache_.find(column_name);
    if (it != cache_.end()) {
      it->second.in_use = false;
    }
  }

  void clear()
  {
    cache_.clear();
  }

 private:
  void evict_oldest()
  {
    auto oldest = cache_.end();
    uint64_t oldest_time = UINT64_MAX;

    for (auto it = cache_.begin(); it != cache_.end(); ++it) {
      if (!it->second.in_use && it->second.last_used < oldest_time) {
        oldest = it;
        oldest_time = it->second.last_used;
      }
    }

    if (oldest != cache_.end()) {
      cache_.erase(oldest);
    }
  }
};
```

### 2.3. 垃圾回收调优

#### 2.3.1. LRU算法优化

```cpp
class OptimizedLRU {
 private:
  struct Node {
    std::string key;
    uint64_t timestamp;
    size_t size;
  };

  std::vector<Node> nodes_;
  size_t current_size_ = 0;
  size_t max_size_ = 100 * 1024 * 1024;  // 100MB

 public:
  void access(const std::string &key, size_t size)
  {
    auto it = std::find_if(nodes_.begin(), nodes_.end(),
                           [&](const Node &n) { return n.key == key; });

    if (it != nodes_.end()) {
      // 更新时间戳
      it->timestamp = get_timestamp();
    } else {
      // 添加新条目
      nodes_.push_back({key, get_timestamp(), size});
      current_size_ += size;

      // 检查是否需要清理
      if (current_size_ > max_size_) {
        cleanup();
      }
    }
  }

  void cleanup()
  {
    // 按时间戳排序
    std::sort(nodes_.begin(), nodes_.end(),
              [](const Node &a, const Node &b) {
                return a.timestamp < b.timestamp;
              });

    // 移除旧条目直到满足大小限制
    size_t target_size = max_size_ * 0.8;  // 保留80%容量
    while (current_size_ > target_size && !nodes_.empty()) {
      current_size_ -= nodes_.back().size;
      nodes_.pop_back();
    }
  }

  uint64_t get_timestamp()
  {
    static std::atomic<uint64_t> counter{0};
    return counter++;
  }
};
```

```cpp
class SmartBatchManager {
 private:
  struct Batch {
    GPUVertBuf *buffer;
    GPUShader *shader;
    int vertex_count;
    GPUPrimType prim_type;
    float4 color;
    bool needs_upload;
  };

  std::vector<Batch> batches_;
  Batch *current_batch_ = nullptr;

 public:
  void begin_batch(GPUShader *shader, GPUPrimType prim, const float4 &color)
  {
    // 查找匹配的批次
    for (auto &batch : batches_) {
      if (batch.shader == shader &&
          batch.prim_type == prim &&
          batch.color == color &&
          batch.vertex_count < 1000) {
        current_batch_ = &batch;
        return;
      }
    }

    // 创建新批次
    Batch new_batch;
    new_batch.shader = shader;
    new_batch.prim_type = prim;
    new_batch.color = color;
    new_batch.vertex_count = 0;
    new_batch.needs_upload = false;

    GPUVertFormat *format = immVertexFormat();
    GPU_vertformat_attr_add(format, "pos", blender::gpu::VertAttrType::SFLOAT_32_32);

    new_batch.buffer = GPU_vertbuf_create_with_format(format);
    GPU_vertbuf_data_alloc(new_batch.buffer, 1000);

    batches_.push_back(new_batch);
    current_batch_ = &batches_.back();
  }

  void add_vertex(float x, float y)
  {
    if (!current_batch_) return;

    GPU_vertbuf_attr_set(current_batch_->buffer, 0,
                        current_batch_->vertex_count, &x, &y);
    current_batch_->vertex_count++;
    current_batch_->needs_upload = true;
  }

  void end_batch()
  {
    if (!current_batch_ || current_batch_->vertex_count == 0) {
      return;
    }

    // 绑定着色器
    GPU_shader_bind(current_batch_->shader);
    immUniformColor4fv(current_batch_->color.data());

    // 上传数据
    if (current_batch_->needs_upload) {
      GPU_vertbuf_use(current_batch_->buffer);
      current_batch_->needs_upload = false;
    }

    // 绘制
    GPU_vertbuf_draw_advanced(
        current_batch_->buffer,
        current_batch_->prim_type,
        0,
        current_batch_->vertex_count
    );

    // 重置
    current_batch_->vertex_count = 0;
    current_batch_ = nullptr;
  }

  void flush_all()
  {
    for (auto &batch : batches_) {
      if (batch.vertex_count > 0) {
        current_batch_ = &batch;
        end_batch();
      }
    }
  }

  ~SmartBatchManager() {
    for (auto &batch : batches_) {
      GPU_vertbuf_discard(batch.buffer);
    }
  }
};
```

```cpp
    range.row_start = scroll_y / drawer.row_height;

        break;
      }
      x_pos += col_w;
```

```cpp
class GPUFilter {
 private:
  GPUShader *compute_shader_;
  GPUTexture *input_texture_;
  GPUTexture *output_texture_;

 public:
  void init()
  {
    // 编译计算着色器
    const char *compute_src = R"(
      #version 430
      layout(local_size_x = 256) in;

      struct Filter {
        int operation;
        float value;
        float threshold;
      };

      layout(std430, binding = 0) buffer Input {
        float data[];
      };

      layout(std430, binding = 1) buffer Output {
        int indices[];
      };

      layout(std430, binding = 2) buffer Filters {
        Filter filters[];
      };

      void main()
      {
        uint idx = gl_GlobalInvocationID.x;
        bool keep = true;

        for (int i = 0; i < filters.length(); i++) {
          Filter f = filters[i];

          if (f.operation == 0) { // EQUAL
            if (abs(data[idx] - f.value) > f.threshold) {
              keep = false;
              break;
            }
          } else if (f.operation == 1) { // GREATER
            if (data[idx] <= f.value) {
              keep = false;
              break;
            }
          } else if (f.operation == 2) { // LESS
            if (data[idx] >= f.value) {
              keep = false;
              break;
            }
          }
        }

        if (keep) {
          indices[idx] = 1;
        } else {
          indices[idx] = 0;
        }
      }
    )";

    compute_shader_ = GPU_shader_create_compute(compute_src);
  }

  void filter(const float *data, int size, const std::vector<Filter> &filters)
  {
    // 上传数据到GPU
    GPU_vertbuf_data_upload(input_buffer_, data, size);

    // 绑定着色器
    GPU_shader_bind(compute_shader_);

    // 设置uniform
    GPU_shader_uniform_int(compute_shader_, "filter_count", filters.size());

    // 分派计算
    int groups = (size + 255) / 256;
    GPU_shader_dispatch(compute_shader_, groups, 1, 1);

    // 读回结果
    GPU_vertbuf_data_read(output_buffer_, indices, size);
  }
};
```

```cpp
class HierarchicalIndex {
 private:
  static const int BLOCK_SIZE = 1024;
  std::vector<std::vector<int>> blocks_;

 public:
  void build(const std::vector<int> &data)
  {
    int num_blocks = (data.size() + BLOCK_SIZE - 1) / BLOCK_SIZE;
    blocks_.resize(num_blocks);

    for (size_t i = 0; i < data.size(); i++) {
      int block_idx = i / BLOCK_SIZE;
      blocks_[block_idx].push_back(data[i]);
    }
  }

  int query(int index) const
  {
    int block_idx = index / BLOCK_SIZE;
    int inner_idx = index % BLOCK_SIZE;

    if (block_idx >= blocks_.size()) {
      return -1;
    }

    if (inner_idx >= blocks_[block_idx].size()) {
      return -1;
    }

    return blocks_[block_idx][inner_idx];
  }

  void range_query(int start, int end, std::vector<int> &result) const
  {
    int start_block = start / BLOCK_SIZE;
    int end_block = end / BLOCK_SIZE;

    for (int block = start_block; block <= end_block; block++) {
      if (block >= blocks_.size()) break;

      int block_start = (block == start_block) ? (start % BLOCK_SIZE) : 0;
      int block_end = (block == end_block) ? (end % BLOCK_SIZE) : blocks_[block].size() - 1;

      for (int i = block_start; i <= block_end; i++) {
        result.push_back(blocks_[block][i]);
      }
    }
  }
};
```

```cpp
class MemoryTracer {
 private:
  struct Allocation {
    void *ptr;
    size_t size;
    std::string tag;
    std::chrono::steady_clock::time_point time;
  };

  std::unordered_map<void*, Allocation> allocations_;
  std::mutex mutex_;
  size_t total_allocated_ = 0;
  size_t peak_usage_ = 0;

 public:
  void *track_alloc(size_t size, const char *tag)
  {
    void *ptr = malloc(size);

    std::lock_guard<std::mutex> lock(mutex_);
    allocations_[ptr] = {
      ptr, size, std::string(tag),
      std::chrono::steady_clock::now()
    };

    total_allocated_ += size;
    peak_usage_ = std::max(peak_usage_, total_allocated_);

    return ptr;
  }

  void track_free(void *ptr)
  {
    if (!ptr) return;

    std::lock_guard<std::mutex> lock(mutex_);
    auto it = allocations_.find(ptr);
    if (it != allocations_.end()) {
      total_allocated_ -= it->second.size;
      allocations_.erase(it);
    }

    free(ptr);
  }

  void print_report()
  {
    std::lock_guard<std::mutex> lock(mutex_);

    printf("\n=== Memory Trace Report ===\n");
    printf("Total Allocated: %.2f MB\n", total_allocated_ / 1024.0 / 1024.0);
    printf("Peak Usage: %.2f MB\n", peak_usage_ / 1024.0 / 1024.0);
    printf("Active Allocations: %zu\n", allocations_.size());

    if (!allocations_.empty()) {
      printf("\nTop Allocations:\n");
      std::vector<Allocation> sorted;
      for (const auto &pair : allocations_) {
        sorted.push_back(pair.second);
      }
      std::sort(sorted.begin(), sorted.end(),
                [](const Allocation &a, const Allocation &b) {
                  return a.size > b.size;
                });

      for (size_t i = 0; i < std::min(size_t(10), sorted.size()); i++) {
        printf("  %s: %.2f KB\n",
               sorted[i].tag.c_str(),
               sorted[i].size / 1024.0);
      }
    }
    printf("===========================\n");
  }

  void check_leaks()
  {
    std::lock_guard<std::mutex> lock(mutex_);

    if (!allocations_.empty()) {
      printf("\n⚠️  MEMORY LEAK DETECTED!\n");
      printf("Leaked allocations: %zu\n", allocations_.size());

      }
    }
  }
```

```cpp
      history_.render_times.erase(history_.render_times.begin());
      history_.draw_calls.erase(history_.draw_calls.begin());

                                          history_.render_times.end());
```

```cpp
   private:
    FunctionProfiler &profiler_;
    std::string name_;
    std::chrono::high_resolution_clock::time_point start_;

   public:
    ScopedCall(FunctionProfiler &profiler, const std::string &name)
        : profiler_(profiler), name_(name)
    {
      if (profiler_.enabled_) {
        start_ = std::chrono::high_resolution_clock::now();
      }
    }

    ~ScopedCall()
    {
      if (profiler_.enabled_) {
        auto end = std::chrono::high_resolution_clock::now();
        std::chrono::duration<double, std::milli> elapsed = end - start_;
        profiler_.record(name_, elapsed.count());
      }
    }
  };

  ScopedCall scoped(const std::string &name)
  {
    return ScopedCall(*this, name);
  }

  void record(const std::string &name, double time_ms)
  {
    auto &info = stats_[name];
    info.name = name;
    info.call_count++;
    info.total_time += time_ms;
    if (time_ms > info.max_time) {
      info.max_time = time_ms;
    }
  }

  void print_report()
  {
    if (!enabled_ || stats_.empty()) return;

    printf("\n=== Function Profile Report ===\n");
    printf("%-30s %10s %10s %10s %10s\n",
           "Function", "Calls", "Total(ms)", "Avg(ms)", "Max(ms)");
    printf("--------------------------------------------------\n");

    // 按总时间排序
    std::vector<CallInfo> sorted;
    for (const auto &pair : stats_) {
      sorted.push_back(pair.second);
    }
    std::sort(sorted.begin(), sorted.end(),
              [](const CallInfo &a, const CallInfo &b) {
                return a.total_time > b.total_time;
              });

    for (const auto &info : sorted) {

             info.call_count,
             info.total_time,
             info.total_time / info.call_count,
             info.max_time);
    }
    printf("===============================\n");
  }

  void reset()
  {
    stats_.clear();
  }

  void enable() { enabled_ = true; }
  void disable() { enabled_ = false; }
};
```

```cpp
class HotspotAnalyzer {
 private:
  struct FunctionCall {
    std::string name;
    double time;
    int depth;
  };

  std::vector<FunctionCall> call_stack_;
  std::vector<FunctionCall> completed_calls_;

 public:
  void enter_function(const std::string &name)
  {
    call_stack_.push_back({name, 0.0, (int)call_stack_.size()});
  }

  void exit_function()
  {
    if (call_stack_.empty()) return;

    FunctionCall call = call_stack_.back();
    call_stack_.pop_back();

    // 记录完成的调用
    completed_calls_.push_back(call);

    // 累加到父函数
    if (!call_stack_.empty()) {
      call_stack_.back().time += call.time;
    }
  }

  void record_time(double time_ms)
  {
    if (!call_stack_.empty()) {
      call_stack_.back().time += time_ms;
    }
  }

  void print_hotspots(int top_n = 10)
  {
    printf("\n=== Hotspot Analysis ===\n");

    // 按时间排序
    std::sort(completed_calls_.begin(), completed_calls_.end(),
              [](const FunctionCall &a, const FunctionCall &b) {
                return a.time > b.time;
              });

    for (int i = 0; i < std::min(top_n, (int)completed_calls_.size()); i++) {
      const auto &call = completed_calls_[i];
      std::string indent(call.depth * 2, ' ');
      printf("%2d. %s%s: %.2f ms\n", i + 1, indent.c_str(),
             call.name.c_str(), call.time);
    }

    printf("=========================\n");
  }

  void reset()
  {
    call_stack_.clear();
    completed_calls_.clear();
  }
};
```

```cpp
class OpenCLFilter {
 private:
  cl_context context_;
  cl_command_queue queue_;
  cl_program program_;
  cl_kernel kernel_;

 public:
  bool init()
  {
    // 初始化OpenCL
    cl_int err;
    context_ = clCreateContextFromType(nullptr, CL_DEVICE_TYPE_GPU, nullptr, nullptr, &err);
    if (err != CL_SUCCESS) return false;

    queue_ = clCreateCommandQueue(context_, device, 0, &err);

    // 编译内核
    const char *source = R"(
      __kernel void filter(__global float *data,
                           __global int *result,
                           float threshold,
                           float value,
                           int size)
      {
        int idx = get_global_id(0);
        if (idx < size) {
          float diff = fabs(data[idx] - value);
          result[idx] = diff < threshold ? 1 : 0;
        }
      }
    )";

    program_ = clCreateProgramWithSource(context_, 1, &source, nullptr, &err);
    clBuildProgram(program_, 0, nullptr, nullptr, nullptr, nullptr);
    kernel_ = clCreateKernel(program_, "filter", &err);

    return true;
  }

  void filter(const std::vector<float> &data,
              float threshold,
              float value,
              std::vector<int> &result)
  {
    cl_int err;
    size_t size = data.size();

    // 创建缓冲区
    cl_mem data_buf = clCreateBuffer(context_, CL_MEM_READ_ONLY,
                                     size * sizeof(float), nullptr, &err);
    cl_mem result_buf = clCreateBuffer(context_, CL_MEM_WRITE_ONLY,
                                       size * sizeof(int), nullptr, &err);

    // 上传数据
    clEnqueueWriteBuffer(queue_, data_buf, CL_TRUE, 0,
                         size * sizeof(float), data.data(), 0, nullptr, nullptr);

    // 设置参数
    clSetKernelArg(kernel_, 0, sizeof(cl_mem), &data_buf);
    clSetKernelArg(kernel_, 1, sizeof(cl_mem), &result_buf);
    clSetKernelArg(kernel_, 2, sizeof(float), &threshold);
    clSetKernelArg(kernel_, 3, sizeof(float), &value);
    clSetKernelArg(kernel_, 4, sizeof(int), &size);

    // 执行
    size_t global_size = size;
    clEnqueueNDRangeKernel(queue_, kernel_, 1, nullptr, &global_size, nullptr, 0, nullptr, nullptr);

    // 读回结果
    clEnqueueReadBuffer(queue_, result_buf, CL_TRUE, 0,
                        size * sizeof(int), result.data(), 0, nullptr, nullptr);

    // 清理
    clReleaseMemObject(data_buf);
    clReleaseMemObject(result_buf);
  }
};
```

```cpp
// 方案1：缓存格式化结果
class FormattedValueCache {
  std::unordered_map<std::string, std::string> cache_;

  const std::string &get(int row, int col, const void *value)
  {
    std::string key = std::to_string(row) + "_" + std::to_string(col);
    auto it = cache_.find(key);
    if (it != cache_.end()) {
      return it->second;
    }
    return cache_[key] = format_value(value, ...);
  }
};

// 方案2：预格式化
void preformat_values(const GVArray &data, std::vector<std::string> &formatted)
{
  formatted.resize(data.size());
  #pragma omp parallel for
  for (int i = 0; i < data.size(); i++) {
    formatted[i] = format_value(data[i], ...);
  }
}

// 方案3：延迟格式化 + 批量
void draw_cells_optimized()
{
  static FormattedValueCache cache;

  for (int row = 0; row < 10000; row++) {
    for (int col = 0; col < 10; col++) {
      const std::string &value = cache.get(row, col, get_cell_value(row, col));
      draw_text(value);
    }
  }
}

// 结果：从5fps提升到60fps
```

### 9.2. 历史数据

#### 9.2.1. 时间序列存储

```cpp
class PerformanceHistory {
 private:
  struct FrameData {
    std::chrono::steady_clock::time_point time;
    SpreadsheetPerformanceMetrics metrics;
  };

  std::vector<FrameData> history_;
  size_t max_history_ = 1000;

 public:
  void record(const SpreadsheetPerformanceMetrics &metrics)
  {
    history_.push_back({
      std::chrono::steady_clock::now(),
      metrics
    });

    if (history_.size() > max_history_) {
      history_.erase(history_.begin());
    }
  }

  void print_trend()
  {
    if (history_.size() < 10) return;

    // 计算最近10帧的平均值
    double avg_render = 0;
    double avg_memory = 0;

    for (size_t i = history_.size() - 10; i < history_.size(); i++) {
      avg_render += history_[i].metrics.render_time_ms;
      avg_memory += history_[i].metrics.memory_usage_bytes;
    }

    avg_render /= 10;
    avg_memory /= 10;

    // 计算趋势（与100帧前比较）
    if (history_.size() >= 100) {
      double old_render = 0;
      for (size_t i = history_.size() - 100; i < history_.size() - 90; i++) {
        old_render += history_[i].metrics.render_time_ms;
      }
      old_render /= 10;

      double change = ((avg_render - old_render) / old_render) * 100.0;

      printf("\nPerformance Trend:\n");
      printf("  Render time: %.2fms (%+.1f%%)\n", avg_render, change);

      if (change > 20.0) {
        printf("  🚨 Performance degradation detected!\n");
      } else if (change < -20.0) {
        printf("  ✅ Performance improved!\n");
      }
    }
  }

  void export_csv(const std::string &filepath)
  {
    FILE *f = fopen(filepath.c_str(), "w");
    if (!f) return;

    fprintf(f, "Time,Render,DrawCalls,Memory,FilterTime\n");

    for (const auto &frame : history_) {
      auto time = std::chrono::duration_cast<std::chrono::milliseconds>(
          frame.time.time_since_epoch()).count();

      fprintf(f, "%lld,%.2f,%d,%.2f,%.2f\n",
              time,
              frame.metrics.render_time_ms,
              frame.metrics.draw_calls_per_frame,
              frame.metrics.memory_usage_bytes / 1024.0 / 1024.0,
              frame.metrics.filter_time_ms);
    }

    fclose(f);
    printf("Exported %zu frames to %s\n", history_.size(), filepath.c_str());
  }
};
```

```cpp
class TieredAlertSystem {
 private:
  enum class Tier {
    NORMAL,
    WARNING,
    CRITICAL,
    EMERGENCY
  };

  Tier current_tier_ = Tier::NORMAL;
  int violation_count_ = 0;

 public:
  void evaluate(const SpreadsheetPerformanceMetrics &metrics)
  {
    Tier new_tier = Tier::NORMAL;

    // 评估指标
    if (metrics.render_time_ms > 16.67) {
      new_tier = Tier::WARNING;
      violation_count_++;
    }

    if (metrics.render_time_ms > 33.33 || metrics.memory_usage_bytes > 500 * 1024 * 1024) {
      new_tier = Tier::CRITICAL;
      violation_count_ += 2;
    }

    if (metrics.render_time_ms > 50.0) {
      new_tier = Tier::EMERGENCY;
      violation_count_ += 5;
    }

    // 状态转换
    if (new_tier != current_tier_) {
      on_tier_change(current_tier_, new_tier);
      current_tier_ = new_tier;
    }

    // 连续违规检查
    if (violation_count_ > 10) {
      printf("🚨 Too many violations! Suggesting restart...\n");
      violation_count_ = 0;
    }
  }

  void on_tier_change(Tier old_tier, Tier new_tier)
  {
    const char *old_name = tier_name(old_tier);
    const char *new_name = tier_name(new_tier);

    printf("\n=== Tier Change: %s -> %s ===\n", old_name, new_name);

    switch (new_tier) {
      case Tier::NORMAL:
        printf("✅ Performance normal\n");
        break;
      case Tier::WARNING:
        printf("⚠️  Performance warning - consider optimization\n");
        break;
      case Tier::CRITICAL:
        printf("🚨 Critical performance issue - immediate action needed\n");
        break;
      case Tier::EMERGENCY:
        printf("🔥 Emergency - system may freeze\n");
        break;
    }
  }

  const char *tier_name(Tier tier)
  {
    switch (tier) {
      case Tier::NORMAL: return "NORMAL";
      case Tier::WARNING: return "WARNING";
      case Tier::CRITICAL: return "CRITICAL";
      case Tier::EMERGENCY: return "EMERGENCY";
      default: return "UNKNOWN";
    }
  }

  void reset()
  {
    current_tier_ = Tier::NORMAL;
    violation_count_ = 0;
  }
};
```
