```cpp
void DRW_text_cache_draw(const DRWTextStore *dt,
                         const ARegion *region,
                         const View3D *v3d,
                         blender::gpu::Texture *depth_tx,
                         const uchar alpha)
{
  if (v3d) {
    blender::gpu::FrameBuffer *depth_frame_buffer = nullptr;

    // ⭐ 新增: 创建深度采样FB
    if (depth_tx) {
      depth_frame_buffer = GPU_framebuffer_create("text_depth_read");
      GPU_framebuffer_texture_attach(depth_frame_buffer, depth_tx, 0, 0);
      GPU_framebuffer_bind(depth_frame_buffer);
    }

    // 遍历所有文本
    while ((vos = ...)) {
      float screen_pos[3];
      ED_view3d_project_float_v3_m4(region, vos->vec, screen_pos, ...);

      bool is_visible = true;

      // ⭐ 关键: 深度遮挡测试
      if (depth_frame_buffer) {
        is_visible = drw_text_depth_test(depth_frame_buffer, screen_pos,
                                         region, vos->vec, text_count);
      }

      if (is_visible) {
        vos->sco[0] = short(screen_pos[0]);
        vos->sco[1] = short(screen_pos[1]);
        tot++;
      }
      else {
        vos->sco[0] = IS_CLIPPED;  // 标记为被遮挡
      }
    }

    // 清理并绘制
    if (depth_frame_buffer) GPU_framebuffer_free(depth_frame_buffer);
    if (tot > 0) drw_text_cache_draw_ex(dt, region, alpha);
  }
}
```

```cpp
void Instance::draw_text(Framebuffer &framebuffer)
{
  if (state.show_text == false) {
    return;
  }
  GPU_framebuffer_bind(framebuffer);

  // ⭐ 新增: 决定是否使用深度测试
  bool use_depth_test = true;

  // X-ray模式特殊处理
  const short xray_mask = V3D_SHADING_XRAY | V3D_SHADING_XRAY_WIREFRAME;
  if (state.v3d->shading.flag & xray_mask) {
    if (state.v3d->shading.type == OB_WIRE ||
        state.v3d->shading.type == OB_SOLID) {
      // X-ray alpha > 0.9 时进行深度测试
      use_depth_test = SHADING_XRAY_ALPHA(state.v3d->shading) > 0.9f;
    }
  }

  // ⭐ 新增: 传递深度纹理
  blender::gpu::Texture *depth_tx =
      use_depth_test ? resources.depth_tx : nullptr;

  // ⭐ 修改调用: 传递depth_tx和alpha
  DRW_text_cache_draw(state.dt, state.region, state.v3d, depth_tx, 255);
}
```

```cpp
void populate_for_geometry(const ObjectRef &ob_ref, const State &state, Manager &manager)
{
    float opacity = state.overlay.viewer_attribute_opacity;
    Object &object = *ob_ref.object;

    switch (object.type) {
        case OB_MESH: {
            Mesh &mesh = DRW_object_get_data_for_drawing<Mesh>(object);

            if (const auto meta_data = mesh.attributes().lookup_meta_data(".viewer")) {
                if (attribute_type_supports_viewer_overlay(meta_data->data_type)) {

                    // ⭐ 新增: 根据用户选择决定显示模式
                    // 需要在State中添加显示模式标志
                    bool show_points = state.overlay.viewer_attribute_mode == V3D_VIEWER_POINTS;
                    bool show_edges = state.overlay.viewer_attribute_mode == V3D_VIEWER_EDGES;

                    gpu::Batch *batch = nullptr;

                    if (show_points) {
                        batch = DRW_cache_mesh_points_viewer_attribute_get(&object);
                    }
                    else if (show_edges) {
                        batch = DRW_cache_mesh_edges_viewer_attribute_get(&object);
                    }
                    else {
                        // 默认: 表面模式
                        batch = DRW_cache_mesh_surface_viewer_attribute_get(&object);
                    }

                    if (batch) {
                        auto &sub = *mesh_sub_;
                        sub.push_constant("opacity", opacity);
                        sub.draw(batch, manager.unique_handle(ob_ref));
                    }
                }
            }
            break;
        }
        // ... 其他类型
    }
}
```

#### 需要的新Shader

**文件**: `overlay_viewer_attribute_vector_vert.glsl`

```glsl
#include "infos/overlay_viewer_attribute_infos.hh"

VERTEX_SHADER_CREATE_INFO(overlay_viewer_attribute_vector)
// 顶点着色器处理矢量到线段的转换
// 基于 overlay_edit_mesh_normal_vert.glsl 修改

#include "draw_model_lib.glsl"
#include "draw_view_clipping_lib.glsl"
#include "draw_view_lib.glsl"

void main()
{
    /*
       输入数据:
       - pos: 顶点位置
       - attribute_value: 视为矢量 (float3) 或颜色+隐含矢量

       需要支持的模式:
       - 表面点 → 箭头
       - 顶点位置 → 箭头
       - 边中点 → 箭头
    */

    float3 start_pos = gpu_attr_load_float3(pos, gpu_attr_0, gl_VertexID / 2);
    float3 vector = load_vector_attribute(gl_VertexID / 2);  // 自定义函数

    // 箭头方向判定 (每两点组成一条线)
    bool is_start = (gl_VertexID & 1) == 0;

    float3 world_pos;
    if (is_start) {
        world_pos = drw_point_object_to_world(start_pos);
    } else {
        // 终点 = 起点 + 矢量
        // 可添加缩放系数让用户控制箭头长度
        float scale = 1.0;  // 可通过uniform传递
        world_pos = drw_point_object_to_world(start_pos + vector * scale);
    }

    // 深度测试 (可选，取决于目标1是否要求)
    if (test_occlusion()) {
        final_color.a = 0.3;  // 被遮挡箭头半透明
    } else {
        final_color = float4(1.0, 0.0, 0.0, 1.0);  // 红色箭头
        // 或使用 attribute_value 作为颜色
    }

    gl_Position = drw_point_world_to_homogenous(world_pos);
}
```
